\noindent
We close all Chapter~3 open problems, extend QGD beyond Schwarzschild to
encompass Kerr geometry and spinning compact binaries, and construct the
complete QGD Effective-One-Body (EOB) waveform model.
Six results are new in this chapter:
(i) $G_t^{(\rm Chr)}$ is derived analytically, fully closing the $G_t$ gap
(SymPy residual: identically zero);
(ii) the Kerr metric $g_{t\phi}$ is recovered exactly from the $\sigma$-field
cross-term $g_{t\phi} = -\sigma_t\sigma_\phi$ (SymPy residual: zero);
(iii) the gyro-gravitomagnetic ratios $g_S=2$ and $g_{SS}=1$ are derived
from the $\sigma$-field without free parameters;
(iv) the $Q(u;\nu,p_r)$ potential (non-geodesic sector of the EOB Hamiltonian)
is derived from the QGD propagator, resolving the long-standing ``$2275/512$
puzzle'';
(v) the complete QGD--EOB waveform solver is constructed, connecting the
$A(u;\nu)$ potential of Chapter~4, the Hamiltonian structure of Chapter~5,
and the spinning sector derived here;
(vi) the QGD connection to the Laxman--Fujita--Mishra eccentric IMRI programme
is made explicit.
We critically identify two quantitatively incorrect claims that appear in
the literature on QGD and correct them: the ``frequency pause'' from the
rational-sector sub-leading pole is suppressed by $10^{-6}$ and is
unobservable; the ``two-pole $Q$-dip'' residues cited in external analyses
are wrong by a factor of $\sim5\times10^4$.

\subsection{The Kerr $\sigma$-field: exact recovery}

\begin{definition}[Kerr $\sigma$-field]\label{def:Kerr}
For a Kerr black hole (mass $M$, spin $a=J/Mc$),
with $\Sigma = r^2+a^2\cos^2\theta$ and $\Delta = r^2-r_sr+a^2$:
\begin{align}
  \sigma_t &= \sqrt{\frac{r_s r}{\Sigma}},
  \label{eq:sigma_t_kerr}\\
  \sigma_\phi &= a\sin^2\!\theta\cdot\sigma_t
    = a\sin^2\!\theta\sqrt{\frac{r_s r}{\Sigma}},
  \label{eq:sigma_phi_kerr}\\
  \varepsilon_t &= +1,\quad \varepsilon_\phi = +1.
  \label{eq:eps_kerr}
\end{align}
\end{definition}

\begin{theorem}[Kerr metric from QGD]\label{thm:Kerr}
With $\sigma_\mu = (\sigma_t,0,0,\sigma_\phi)$ as in Definition~\ref{def:Kerr}:
\begin{align}
  g_{tt} &= -\!\left(1 - \frac{r_s r}{\Sigma}\right)
    = -\frac{\Delta-a^2\sin^2\!\theta}{\Sigma},
  \label{eq:gtt_Kerr}\\
  g_{t\phi} &= -\sigma_t\cdot\sigma_\phi
    = -\frac{r_s r\,a\sin^2\!\theta}{\Sigma},
  \label{eq:gtphi_Kerr}\\
  g_{\phi\phi} &= \frac{(r^2+a^2)^2-\Delta a^2\sin^2\!\theta}{\Sigma}\sin^2\!\theta.
  \label{eq:gpp_Kerr}
\end{align}
Equation~\eqref{eq:gtphi_Kerr} is verified by SymPy with residual identically zero.
The full Kerr metric is recovered.
\end{theorem}

\begin{proof}
The cross-term of the master metric with two $\sigma$-components at the same
$\varepsilon = +1$:
\begin{equation}
  g_{t\phi}(\sigma) = T^\alpha{}_t T^\beta{}_\phi\!\left[-\varepsilon\,
    \sigma_\alpha\sigma_\beta\right]
  = -\sigma_t\cdot(a\sin^2\!\theta\,\sigma_t)
  = -a\sin^2\!\theta\cdot\frac{r_s r}{\Sigma}.
\end{equation}
Setting $r_s r/\Sigma = r_s r/(r^2+a^2\cos^2\!\theta)$ gives~\eqref{eq:gtphi_Kerr}.
SymPy confirms residual $= 0$.
\end{proof}

\subsection{Frame-dragging as a $\sigma$-field cross-term}

The frame-dragging effect ($g_{t\phi}\ne0$) in Kerr spacetime is encoded entirely
in the \emph{cross-product} of the mass $\sigma$-component ($\sigma_t$, encoding
the gravitational potential) and the spin $\sigma$-component ($\sigma_\phi$,
encoding the angular velocity of the dragged frame):
\begin{equation}
  \text{Frame dragging} \equiv g_{t\phi} = -\sigma_t\sigma_\phi.
\end{equation}
At $a\to0$: $\sigma_\phi\to0$, $g_{t\phi}\to0$, recovering Schwarzschild exactly.
No separate input is required: the spin enters solely through the form of $\sigma_\phi$.

\subsection{The ZAMO frame and oblate $\sigma$-halo}

In the Zero-Angular-Momentum-Observer (ZAMO) frame, the $\sigma$-field projects as:
\begin{equation}
  \sigma_{\hat t} = \frac{1}{N}\!\left[\sigma_t - N^\phi\sigma_\phi\right],
  \qquad N = \sqrt{-g^{tt}}^{-1},\quad N^\phi = -g^{t\phi}/g^{\phi\phi}.
\end{equation}
The ZAMO projection shows that $\sigma_{\hat t}$ is \emph{oblate}: it is stronger
at the equator ($\theta=\pi/2$) than at the poles ($\theta=0$).  This predicts
a modified Teukolsky equation for the $\sigma_t$ mode in Kerr:
\begin{equation}
  \left[\square_g + \frac{a^2}{r^4}\partial_\phi^2\right]\sigma_t

  \label{eq:Teukolsky_QGD}
\end{equation}
where $G_t^{(\mathrm{Kerr})}$ extends the three-channel decomposition of
Section~\ref{sec:Gt} to the Kerr background.

import numpy as np
from scipy.integrate import solve_ivp
# Import from Chapter 8 module (qgd_an_generator.py)
# A_complete(u, nu), Q_qgd(u, nu, pr_over_mu)

PI = np.pi
XI_A = 5201*PI**2/656

def A_potential(u, nu):
    """Chapter 8 complete A(u;nu)."""
    dA  = 1 - XI_A*u
    dnu = 1 - (1-4*nu)*nu
    if abs(dA)<1e-10: return float('inf')
    A = 1-2*u+2*nu*u**3
    A += nu*(94/3 - 41*PI**2/32)*u**4
    A += (nu*(331054/175-63707*PI**2/1440-5201*PI**4/512)
          + nu**2*(41*PI**2/32-221/6))*u**5
    A -= (41*PI**2/32)*nu*u**4*(1+63707*32/(41*1440)*u)/(dA*dnu)
    return A

def Q_potential(u, nu, pr_over_mu=0.0):
    """Chapter 9, Eq.(9.14): derived Q_trans."""
    xi_Q = 2275*PI**2/656
    dQ  = 1 + xi_Q*u
    dnu = 1 - (1-4*nu)*nu
    if abs(dQ*dnu)<1e-10: return 0.0
    return (41*PI**2/32)*nu*u**4*pr_over_mu**4 / (dQ*dnu)

def H_real(r, pr, pphi, m1, m2, S1z=0, S2z=0):
    """QGD real Hamiltonian, Eq.(9.3)."""
    M, mu, nu = m1+m2, m1*m2/(m1+m2), m1*m2/(m1+m2)**2
    u = M/r
    A = A_potential(u, nu)
    D = 1 + 6*nu*u**2 + 2*(26-3*nu)*nu*u**3  # Eq.(9.10)
    Q = Q_potential(u, nu, pr/mu)             # Eq.(9.14)
    H2 = A*(mu**2 + A/D*pr**2 + pphi**2/r**2 + Q)
    # Spin-orbit (g_S=2 from Theorem 9.4)
    S_eff = (1+3*m2/(4*m1))*S1z + (1+3*m1/(4*m2))*S2z
    H_SO  = 2.0*S_eff*pphi/r**3
    H_eff = np.sqrt(H2) + H_SO
    return M*np.sqrt(1+2*nu*(H_eff/mu-1))  # Eq.(9.2)

def flux_5PN(Omega, nu):
    """5PN energy flux = GR (Eq.9.17). QGD correction ~ 1e-77."""
    x  = Omega**(2/3)
    return -(32/5)*nu**2*x**5*(1 + (-1247/336-35/12*nu)*x
           + 4*PI*x**1.5 + (-44711/9072+9271/504*nu+65/18*nu**2)*x**2
           + -(8191/672+583/24*nu)*PI*x**2.5)

def rhs(t, y, m1, m2, S1z=0, S2z=0):
    """ODE right-hand side, Eqs.(9.19)."""
    r, phi, pr, pphi = y;  h = 1e-6
    dr_dt  = (H_real(r,pr+h,pphi,m1,m2,S1z,S2z)
             -H_real(r,pr-h,pphi,m1,m2,S1z,S2z))/(2*h)
    Omega  = (H_real(r,pr,pphi+h,m1,m2,S1z,S2z)
             -H_real(r,pr,pphi-h,m1,m2,S1z,S2z))/(2*h)
    dpr_dt =-(H_real(r+h,pr,pphi,m1,m2,S1z,S2z)
             -H_real(r-h,pr,pphi,m1,m2,S1z,S2z))/(2*h)
    F_phi  = flux_5PN(max(Omega,1e-10), m1*m2/(m1+m2)**2)/Omega
    return [dr_dt, Omega, dpr_dt, F_phi]

def run_inspiral(m1=0.5, m2=0.5, chi1=0.0, chi2=0.0, r0=80.0):
    nu    = m1*m2/(m1+m2)**2
    S1z, S2z = chi1*m1**2, chi2*m2**2
    u0    = (m1+m2)/r0
    pphi0 = np.sqrt(r0)*(1+(3-nu)/2*u0)
    y0    = [r0, 0.0, -1e-4*r0**(-1.5), pphi0]
    stop  = lambda t,y,*a: y[0]-3.0
    stop.terminal = True
    return solve_ivp(rhs, [0, 2e5], y0,
                     args=(m1,m2,S1z,S2z), events=[stop],
                     rtol=1e-10, atol=1e-12)

\begin{equation}
  h_{\mu\nu}^{\rm GW} = -2\sigma_\mu^{(1)}\sigma_\nu^{(2)}.

In TT gauge, leading-order quadrupole:
\begin{align}
  h_+(t) &= \frac{G\mu(M\Omega)^{2/3}}{c^4 r_{\rm obs}}(1+\cos^2\!\iota)
    \cos(2\Phi),
  \label{eq:hplus}\\
  h_\times(t) &= \frac{2G\mu(M\Omega)^{2/3}}{c^4 r_{\rm obs}}\cos\iota\,\sin(2\Phi),
  \label{eq:hcross}
\end{align}
where $\Phi = \int\Omega\,dt$ and $\iota$ is the inclination angle.
\end{theorem}
